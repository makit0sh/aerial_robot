cmake_minimum_required(VERSION 2.8.3)
project(hydrus)

## Find catkin macros and libraries
## if COMPONENTS list like find_package(catkin REQUIRED COMPONENTS xyz)
## is used, also find other catkin packages
find_package(catkin REQUIRED COMPONENTS
  roscpp
  aerial_robot_msgs
  aerial_robot_control
  aerial_robot_model
  spinal
  eigen_conversions
  tf_conversions
  std_srvs
  dynamic_reconfigure
  )

find_package(Eigen3 REQUIRED)
# TODO Try building in catkin
find_library(
  RBDL_LIBRARY
  REQUIRED
  NAMES rbdl
  PATHS /usr/local/lib
  PATH_SUFFIXES lib
  NO_DEFAULT_PATH
  )
find_library(
  RBDL_URDFReader_LIBRARY
  REQUIRED
  NAMES rbdl_urdfreader
  PATHS /usr/local/lib
  PATH_SUFFIXES lib
  NO_DEFAULT_PATH
  )

SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++14")

generate_dynamic_reconfigure_options(
  cfg/LQI.cfg
  cfg/LQI_torsion.cfg)

catkin_package(
  INCLUDE_DIRS include test
  LIBRARIES hydrus_robot_model hydrus_controller_pluginlib hydrus_numerical_jacobians
  CATKIN_DEPENDS roscpp aerial_robot_control aerial_robot_model aerial_robot_msgs spinal tf_conversions eigen_conversions
  DEPENDS EIGEN3
)

# Eigen requires optimization to get good performance
# http://eigen.tuxfamily.org/index.php?title=FAQ#Optimization
if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE RelWithDebInfo)
endif()
set(CMAKE_CXX_FLAGS_RELWITHDEBINFO "-O3 -g -DNDEBUG")
